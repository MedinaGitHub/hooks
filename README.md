## Install

```bash
npm install --save example-hook
```

## Usage

```jsx
import React, { Component } from 'react'

import { useMyHook } from 'example-hook'

const Example = () => {
  const example = useMyHook()
  return (
    <div>{example}</div>
  )
}
```
